# Clean .htaccess for Laravel on cPanel
RewriteEngine On

# Redirect all requests to the public folder
RewriteCond %{REQUEST_URI} !^/public/
RewriteRule ^(.*)$ public/$1 [L]

# Prevent access to sensitive files
<Files .*>
    Require all denied
</Files>

# Allow access to public folder
<Files "public/*">
    Require all granted
</Files>
